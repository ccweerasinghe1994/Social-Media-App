/**
 * Create a single new user
 * @async
 * @param {object} user
 * @returns {object|void}
 */
const create = async (user) => {
  try {
    let response = await fetch("/api/users/", {
      method: "POST",
      headers: {
        Accept: "application/json",
        "Content-Type": "application/json",
      },
      body: JSON.stringify(user),
    });

    return await response.json();
  } catch (error) {
    console.log(error);
  }
};
